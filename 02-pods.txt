# Create a new pod
kubectl create -f pods/hello-app-pod.yaml -n mayuran-dev-space

# Delete pod
kubectl delete -f pods/hello-app-pod.yaml -n mayuran-dev-space

# List pods
kubectl get pods -A

# Access app with port forwarding
kubectl port-forward hello-app-pod 8080:8080
